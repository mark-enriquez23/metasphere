  /**
   * Gets the Current Date
   *
   * @returns string - Current Date with the format given
   */
  getDate(): string {
    var date = new Date("Sun May 11,2014");
    var dateString = new Date(date.getTime() - (date.getTimezoneOffset() * 60000 ))
                        .toISOString()
                        .split("T")[0];
    return dateString;
  }
